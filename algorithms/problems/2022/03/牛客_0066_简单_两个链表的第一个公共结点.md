class Solution:
    def FindFirstCommonNode(self , pHead1 , pHead2 ):

        def get_len(cur):
            cnt = 0
            while cur:
                cnt += 1
                cur = cur.next
            return cnt

        l1 = get_len(pHead1)
        l2 = get_len(pHead2)

        if l1 < l2:
            return self.FindFirstCommonNode(pHead2, pHead1)

        cur1, cur2 = pHead1, pHead2

        d = l1 - l2
        while d:
            cur1 = cur1.next
            d -= 1

        while cur1 != cur2:
            cur1 = cur1.next
            cur2 = cur2.next

        return cur1
