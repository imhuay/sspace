    def GetLeastNumbers_Solution(self, tinput, k):

        def partition(a, lo, hi):
            if lo >= hi: return

            p = a[lo]
            l, r = lo, hi
            while l < r:
                while l < r and a[r] >= p: r -= 1
                while l < r and a[l] <= p: l += 1
                a[l], a[r] = a[r], a[l]
