class Solution:
    def Serialize(self, root):
        if not root: return ''

        from collections import deque
        q = deque([root])
        ret = []
        while q:
            p = q.popleft()
            if p:
                ret.append(str(p.val))
                q.append(p.left)
                q.append(p.right)
            else:
                ret.append('#')  # æ·»åŠ ç©ºèŠ‚ç‚¹

        return ','.join(ret)

        from collections import deque
        data = s.split(',')
        i = 0  # ç´¢å¼•æŒ‡é’ˆ
        root = TreeNode(int(data[i]))
        i += 1

            if data[i] != '#':
                node.left = TreeNode(int(data[i]))
                q.append(node.left)
            i += 1

            if data[i] != '#':
                node.right = TreeNode(int(data[i]))
                q.append(node.right)
            i += 1
