class Solution:
    def isPail(self , head: ListNode) -> bool:
        import sys
        sys.setrecursionlimit(1000001)
        
        self.forward = head
        
        def dfs(backward):
            if not backward: 
                return True
            r1 = dfs(backward.next)
            r2 = backward.val == self.forward.val
            self.forward = self.forward.next
            return r1 and r2
        
        return dfs(head)

class Solution:

        tmp = []
        cur = head
        while cur:
            tmp.append(cur.val)
            cur = cur.next
        
        N = len(tmp)
