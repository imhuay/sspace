class Solution:
    def longestPalindrome(self, s: str) -> str:

        n = len(s)
        dp = [[0] * n for _ in range(n)]

        for i in range(n):
            dp[i][i] = 1

        start = 0
        length = 1
        for j in range(1, n):  # å­ä¸²çš„ç»“æŸä½ç½®
            for i in range(j - 1, -1, -1):  # å­ä¸²çš„å¼€å§‹ä½ç½®

                        start = i

        return s[start: start + length]
