```python
class Solution:
    def maxArea(self, height: List[int]) -> int:

        def cur_amount():
            return (r - l) * min(height[l], height[r])

        l, r = 0, len(height) - 1
        ret = cur_amount()
        while l < r:
            if height[l] < height[r]:
                l += 1
            else:
                r -= 1

            ret = max(ret, cur_amount())

        return ret
```
